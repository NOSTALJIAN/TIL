실습

calc.js

//  계산 기능을 하는 파일

const add = (a, b) => a + b;
const sub = (a, b) => a - b;

module.exports = {
  moduleName: 'calc module',
  add: add,
  sub: sub,
};

index.js

//  다른 파일인 calc.js의 데이터를 불러와서 사용

const calc = require('./calc');

console.log(calc.add(1, 2));
console.log(calc.add(4, 5));
console.log(calc.sub(10, 2));
