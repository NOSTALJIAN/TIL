const getMeal = (mealType) => {
  if (mealType === "한식") return "불고기";
  if (mealType === "양식") return "파스타";
  if (mealType === "중식") return "멘보샤";
  if (mealType === "일식") return "초밥";
  return "굶기";
};
console.log(getMeal("한식")); //  불고기
console.log(getMeal("중식")); //  멘보샤

동기 & 비동기

동기 방식의 처리

자바스크립트의 싱글 스레드 작업 수행 방식

  • 자바스크립트는 코드가 작성된 순서대로 작업을 처리함
  • 이전 작업이 진행 중 일 때는 다음 작업을 수행하지 않고 기다림
  • 먼저 작성된 코드를 먼저 다 실행하고 나서 뒤에 작성된 코드를 실행함
  • 블로킹 방식
/* 수행해야 할 작업(함수) 3개 */
function taskA() {
  console.log("TASK A");
}
function taskB() {
  console.log("TASK B");
}
function taskC() {
  console.log("TASK C");
}

/* 작업 실행(함수 호출) 순서 */
taskA();
taskB();
taskC();

동기처리 방식의 문제점

  • 하나의 작업이 너무 오래 걸리게 될 시, 모든 작업이 오래 걸리는 하나의 작업이 종료되기 전 까지 올 스탑 되기 때문에 전반적인 흐름이 느려진다.

    ex) 버튼 하나하나마다 30초씩 걸리면….. 🤯

비동기 방식의 처리

멀티 쓰레드

  • 코드를 실행하는 일꾼 Thread를 여러 개 사용하는 방식인 MultiThread 방식으로 작동시키면 작업 분할이 가능

    ➠ 작업이 오래 걸려도 다른 일꾼 Thread에게 지시하면 되므로 괜찮음

  • 자바스크립트는 싱글 쓰레드로 동작함

    ➠ 일꾼을 여러 개 사용하는 방식이 불가

  • 싱글 쓰레드 방식을 이용하면서, 동기적 작업의 단점을 극복하기 위해 여러개의 작업을 동시에 실행시킴

    ➠ 먼저 작성된 코드의 결과를 기다리지 않고 다음 코드를 바로 실행

    ➠ 비동기 작업(논 블로킹 방식)

function taskA() {
  console.log("TASK A");
}
function taskB() {
  console.log("TASK B");
}
function taskC() {
  console.log("TASK C");
}

taskA((resultA) => {
  console.log(`A 끝났습니다. 작업 결과 : ${resultA}`);  //  A 콜백
});
taskB((resultB) => {
  console.log(`B 끝났습니다. 작업 결과 : ${resultB}`);  //  B 콜백
});
taskC((resultC) => {
  console.log(`C 끝났습니다. 작업 결과 : ${resultC}`);  //  C 콜백
});

활용

기존

/* 기존 동기방식 처리 */
function taskA() {
  console.log("A 작업 끝");
}

taskA();
console.log("코드 끝"); //  코드 끝

비동기 방식 처리 1

function taskA() {
  setTimeout(() => {
    console.log("A TASK END");
  }, 2000); //  2초 후에 출력
}

taskA();
console.log("코드 끝"); //  코드 끝, A TASK END

비동기 방식 처리 2

function taskA(a, b, cb) {
  setTimeout(() => {
    const res = a + b; //  지역 상수
    cb(res); //  콜백 함수
  }, 3000); //  3초 후에 출력
}
function taskB(a, cb) {
  setTimeout(() => {
    const res = a * 2;
    cb(res);
  }, 1000);
}
function taskC(a, cb) {
  setTimeout(() => {
    const res = a * -1;
    cb(res);
  }, 2000);
}

taskA(3, 4, (res) => {
  console.log("A TASK RESULT : ", res);
});
taskB(7, (res) => {
  console.log("B TASK RESULT : ", res);
});
taskC(14, (res) => {

활용 2 - 비동기 방식

예제 1

function asyncAdd(a, b, cb) {
  setTimeout(() => {
    const res = a + b;
    cb(res);
  }, 3000); //  3초 후
}

asyncAdd(1, 3, (res) => {
  console.log("결과  : ", res); //  결과 : 4
});
  1. Main Context가 Call Stack에 들어오게 됨(실행)
  1. asyncAdd() 실행, Call Stack에 추가
  1. setTimeout(cb()) 실행, Call Stack에 추가
  1. Web APIs로 setTimeout(cb()) 넘김 ➠ 멈추는게 아님
  1. asyncAdd() 종료, Call Stack에서 제거
  1. setTimeout(cb())의 cb()가 Callback Queue로 옮겨짐
  1. Event Loop는 Call Stack에 Main Context를 제외한 다른 함수가 남아있는지 확인
  1. Main Context를 제외한 다른 함수가 없으면 Event Loop가 cb()를 Call Stack로 옮김
  1. cb() 실행, Call Stack에서 제거
  1. Main Context가 Call Stack에서 제거 됨(종료)

Promise

과정

비동기 작업이 가질 수 있는 3가지 상태

  1. Pending(대기 상태)
  1. Fulfilled(성공)
  1. Rejected(실패)

Resolve - 해결

❓ Pending 상태에서 Fulfilled로 변화하는 과정

Reject - 거부

❓ Pending 상태에서 Rejected로 변화하는 과정

실습

예제 1

function isPositive(number, resolve, reject) {
  setTimeout(() => {
    if (typeof number === "number") {
      //  성공 -> resolve
      resolve(number >= 0 ? "양수" : "음수");
    } else {
      //  실패 -> reject
      reject("주어진 값이 숫자형 값이 아닙니다.");
    }
  }, 2000);
}

isPositive(
  10,
  (res) => {
    console.log("성공적으로 수행됨 : ", res);
  },
  (err) => {
    console.log("실패 하였음 : ", err);
  }
);  //  성공적으로 수행됨 : 양수

isPositive(
  [],
  (res) => {
    console.log("성공적으로 수행됨 : ", res);
  },
  (err) => {
    console.log("실패 하였음 : ", err);
  }
);  //  실패 하였음 :  주어진 값이 숫자형 값이 아닙니다.

예제 1 - Promise 사용

function isPositiveP(number) {
  const executor = (resolve, reject) => {
    setTimeout(() => {
      if (typeof number === "number") {
        //  성공 -> resolve
        console.log(number);
        resolve(number >= 0 ? "양수" : "음수");
      } else {
        //  실패 -> reject
        reject("주어진 값이 숫자형 값이 아닙니다.");
      }
    }, 2000);
  };
  const asyncTask = new Promise(executor);
  return asyncTask;
}

const res = isPositiveP(101);
res
  .then((res) => {
    //  resolve
    console.log("작업 성공 : ", res);
  })
  .catch((err) => {
    //  reject
    console.log("작업 실패 : ", err);
  });
//  101, 작업 성공 :  양수

const res = isPositiveP([]);
res
  .then((res) => {
    //  resolve
    console.log("작업 성공 : ", res);
  })
  .catch((err) => {
    //  reject
    console.log("작업 실패 : ", err);
  });
//  작업 실패 :  주어진 값이 숫자형 값이 아닙니다.
