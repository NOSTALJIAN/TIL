변수

let age = 25;
console.log(age); // 25

age = 30;
console.log(age); // 30

let ag_$e = 25;
console.log(ag_$e); // 25
  • 변수명에는 기호 사용 불가 (_, $ 제외)
  • 변수명 앞에는 숫자 사용 불가

var age = 25;
console.log(age); // 25

age = 30;
console.log(age); // 30
  • var 와 let 은 사용법은 같음
  • var 는 변수를 중복선언할 수 없음

상수

자료형

Primitive Type(원시타입)

  • 한번에 하나의 값 만 가질 수 있음
  • 하나의 고정된 저장 공간 이용
  • 종류
    1. Number
    1. String
    1. Boolean
    1. Undefined
    1. Null

비교 연산자

let compareA = 1 != "1";

let compareA = 1 !== "1";
console.log(compareA); // true, 1과 "1"의 값과 문자열이 같지 않은지 비교

함수 사용

function getArea() {
  let width = 10;
  let heigh = 20;

  let area = width * heigh;
  console.log(area);
} // 함수 선언식, 함수 선언 방식의 함수 생성

getArea(); // 200
console.log("함수 실행 완료");

콜백 함수

function checkMood(mood, goodCallback, badCallback) {
  if (mood === "good") {
    // 기분이 좋을 때 하는 동작
    goodCallback();
  } else {
    // 기분이 안 좋을 때ㅐ 하는 동작
    badCallback();
  }
}

function cry() {
  console.log("ACTION :: CRY");
}

function sing() {
  console.log("ACTION :: SING");
}

function dance() {
  console.log("ACTION: DANCE");
}

// "sad"일 떄(기분이 안 좋을 떄), badCallback에 대입되고 cry() 실행
checkMood("sad", sing, cry); //  ACTION :: CRY

let person = {
  name: "서지안",
  age: 29,
  tall: 171
};

//  key 값만 뽑아서 반환시키는 내장함수
const personKeys = Object.keys(person);
console.log(personKeys); //  {"name", "age", "tall"}
for (let i = 0; i < personKeys.length; i++) {
  console.log(personKeys[i]); //  name, age, tall

  const curKey = personKeys[i];
  const curValue = person[curKey];
  console.log(`${curKey}: ${curValue}`); //  name: 서지안, age: 29, tall: 171
}

//  value 값만 뽑아서 반환시키는 내장함수
const personValues = Object.values(person);
for (let i = 0; i < personValues.length; i++) {
  console.log(personValues[i]); //  서지안, 29, 171
}

Sort

//  숫자는 한번에 제대로 나열이 안됨
let numbers = [0, 1, 3, 2, 10, 30, 20];
numbers.sort();
console.log(numbers); //  [0, 1, 10, 2, 20, 3, 30]

//  오름차순
const compare = (a, b) => {
  if (a > b) {
    //  1. 크다
    return 1;
  }
  if (a < b) {
    //  2. 작다
    return -1;
  }
  //  3. 같다
  return 0;
};
numbers.sort(compare);
console.log(numbers); //  [0, 1, 2, 3, 10, 20, 30]

//  내림차순
const compare = (a, b) => {
  if (a > b) {
    //  1. 크다
    return -1;
  }
  if (a < b) {
    //  2. 작다
    return 1;
  }
  //  3. 같다
  return 0;
};
numbers.sort(compare);
console.log(numbers); //  [30, 20, 10, 3, 2, 1, 0]

Join

const arr = ["서지안", "님", "안녕하세요", "또 오셨군요"];

console.log(arr); //  ["서지안", "님", "안녕하세요", "또 오셨군요"]
console.log(arr[0], arr[1], arr[2], arr[3]);  //  서지안 님 안녕하세요 또 오셨군요

console.log(arr.join()); //  서지안,님,안녕하세요,또 오셨군요
console.log(arr.join(" ")); //  서지안 님 안녕하세요 또 오셨군요
console.log(arr.join("바보")); //  서지안바보님바보안녕하세요바보또 오셨군요
console.log(arr.join("===")); //  서지안===님===안녕하세요===또 오셨군요
